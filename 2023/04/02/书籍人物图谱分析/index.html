定义命名实体识别任务

import os
from tqdm import tqdm
from paddlenlp import Taskflow

# 定义命名实体识别任务
ner = Taskflow("ner")

# 读取文本文件中的内容
with open("故乡天下黄花.txt", "r", encoding="utf-8") as f:
    text = f.read()

# 将文本文件分割成100份
num_parts = 100
part_size = len(text) // num_parts
parts = [text[i:i+part_size] for i in range(0, len(text), part_size)]

# 对每份文件进行命名实体识别，并将识别结果添加到实体集合中
entities = set()
for i, part in tqdm(enumerate(parts), total=num_parts):
    result = ner(part)
    for item in result:
        if item[1] != "O":
            entities.add((item[0], item[1]))

# 将实体集合转换为字符串，并存储到文件中
with open("ner_result.txt", "w", encoding="utf-8") as f:
    entity_result_str = "\n".join(
        ["{}\t{}".format(item[0], item[1]) for item in entities]
    )
    f.write(entity_result_str)

绘制图谱网络

# 将人物共现的语句筛选获取，并生成图谱网络；
import re

import networkx as nx
from pyecharts import options as opts
from pyecharts.charts import Graph

# 定义集合和文本
keywords = merged_words

# 读取txt文件

# 将字符串分割成句子
sentences = re.split("[。！？；]\n*", text_re)  # 以句号、感叹号、问号和分号作为分割符

# 创建无向图
G = nx.Graph()

# 添加节点
for keyword in keywords:
    G.add_node(keyword)

# 统计共现次数，并添加边
for keyword1 in keywords:
    for keyword2 in keywords:
        if keyword1 != keyword2:
            count = 0
            sents = []
            for sentence in sentences:
                if keyword1 in sentence and keyword2 in sentence:
                    count += 1
                    sents.append(sentence)
            if count > 0:
                G.add_edge(keyword1, keyword2, weight=count, sents=sents)

# 获取边的权重信息和所在句子信息
edge_widths = nx.get_edge_attributes(G, "weight")
edge_sents = nx.get_edge_attributes(G, "sents")

# 将节点和边转换为pyecharts所需的格式
nodes = [{"name": keyword} for keyword in keywords]
links = []
for keyword1, keyword2, count in G.edges(data="weight"):
    link = {"source": keyword1, "target": keyword2, "value": count}
    sents = edge_sents[(keyword1, keyword2)]
    if sents:
        link["tooltip"] = {"formatter": "<br>".join(sents)}
    links.append(link)

# 绘制图形
graph = Graph(
    init_opts=opts.InitOpts(
        width="1800px",
        height="900px",
    )
)
graph.add(
    "",
    nodes,
    links,
    repulsion=10000,
    label_opts=opts.LabelOpts(is_show=True, position="right", font_size=12),
    linestyle_opts=opts.LineStyleOpts(curve=0.2),
    itemstyle_opts=opts.ItemStyleOpts(opacity=0.5),
)
graph.set_global_opts(title_opts=opts.TitleOpts(title="故乡天下黄花_人物关系图"))
graph.render("故乡天下黄花_人物关系图.html")
