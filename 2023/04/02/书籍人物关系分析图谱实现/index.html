# 将人物共现的语句筛选获取，并生成图谱网络；
import re

import networkx as nx
from pyecharts import options as opts
from pyecharts.charts import Graph

# 定义集合和文本
keywords = merged_words

# 读取txt文件

# 将字符串分割成句子
sentences = re.split("[。！？；]\n*", text_re)  # 以句号、感叹号、问号和分号作为分割符

# 创建无向图
G = nx.Graph()

# 添加节点
for keyword in keywords:
    G.add_node(keyword)

# 统计共现次数，并添加边
for keyword1 in keywords:
    for keyword2 in keywords:
        if keyword1 != keyword2:
            count = 0
            sents = []
            for sentence in sentences:
                if keyword1 in sentence and keyword2 in sentence:
                    count += 1
                    sents.append(sentence)
            if count > 0:
                G.add_edge(keyword1, keyword2, weight=count, sents=sents)

# 获取边的权重信息和所在句子信息
edge_widths = nx.get_edge_attributes(G, "weight")
edge_sents = nx.get_edge_attributes(G, "sents")

# 将节点和边转换为pyecharts所需的格式
nodes = [{"name": keyword} for keyword in keywords]
links = []
for keyword1, keyword2, count in G.edges(data="weight"):
    link = {"source": keyword1, "target": keyword2, "value": count}
    sents = edge_sents[(keyword1, keyword2)]
    if sents:
        link["tooltip"] = {"formatter": "<br>".join(sents)}
    links.append(link)

# 绘制图形
graph = Graph(
    init_opts=opts.InitOpts(
        width="1800px",
        height="900px",
    )
)
graph.add(
    "",
    nodes,
    links,
    repulsion=10000,
    label_opts=opts.LabelOpts(is_show=True, position="right", font_size=12),
    linestyle_opts=opts.LineStyleOpts(curve=0.2),
    itemstyle_opts=opts.ItemStyleOpts(opacity=0.5),
)
graph.set_global_opts(title_opts=opts.TitleOpts(title="手机_人物关系图"))
graph.render("手机_人物关系图.html")
